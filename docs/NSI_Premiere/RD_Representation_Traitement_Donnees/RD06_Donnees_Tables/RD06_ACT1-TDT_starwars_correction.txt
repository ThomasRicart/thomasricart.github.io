# Question 7: Les 20 personnages les plus grands
# On filtre les valeurs numériques valides avant de trier
valides_taille = []
for p in dataset:
    if p['height'] != 'unknown' and p['height'] != 'NA':
        valides_taille.append(p)

top_20_grands = sorted(valides_taille, key=lambda p: float(p['height']), reverse=True)[:20]

# Question 9: 10 personnages humains de Tatooine les plus grands

        if p['height'] != 'unknown' and p['height'] != 'NA':
            selection.append(p)

top_10_tatooine = sorted(selection, key=lambda p: float(p['height']), reverse=True)[:10]
