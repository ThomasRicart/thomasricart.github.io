\begin{definition}[Loi normale]{}
	
 Une variable aléatoire X suit une loi normale d'espérance $\mu$ et d'écart-type $\sigma$, notée $\mathcal{N}(\mu;\sigma)$, lorsque la densité de probabilité $f$ est définie par:
\[f(x)=\dfrac{1}{\sigma\sqrt{2\pi}}e^{-\dfrac{1}{2}\left(\dfrac{x-\mu}{\sigma}\right)^2}\]

\begin{center}
\includegraphics[width = 0.9\linewidth]{loi_normale2.png}
\end{center}
\end{definition}

\begin{itemize}
	\item[$\bullet$] 	La courbe représentative de la densité de probabilité \og en cloche \fg{}  est appelée une courbe de Gauss. Elle admet un axe de symétrie d'équation $x=\mu$.

	\item[$\bullet$] L'adjectif « normale » s'explique par le fait que cette loi décrit et modélise la majorité des situations concrètes dans lesquelles le résultat dépend de nombreux paramètres aléatoires indépendants.

\end{itemize}

\begin{propriete}[Caractères de position et de dispersion]{}

	$X \sim \mathcal{N}(\mu;\sigma)$: Espérance $E(X)=\mu$ et écart-type $\sigma(X)=\sigma$.

\end{propriete}

\begin{minipage}{0.55\linewidth}

On remarque que:

\begin{itemize}
	\item[$\bullet$] 	$\mu$ est un caractère de position. Lorsque $\mu$ varie l'extremum de la courbe de déplace le long de l'axe des abscisses. 

	\item[$\bullet$] $\sigma$ est un caractère de dispersion. Lorsque $\sigma$ augmente, la répartition est plus étalée.

\end{itemize}
	


\end{minipage}
\begin{minipage}{0.45\linewidth}
\includegraphics[width = 0.9\linewidth]{musigma.png}
\end{minipage}


\bigskip

\begin{propriete}[Règle empirique]{}
Soit $X$ une variable aléatoire suivant la loi normale $\mathcal{N}(\mu;\sigma)$, alors:

\begin{minipage}{0.5\linewidth}
\[P(\mu-\sigma\le X\le\mu+\sigma)\approx 0,68\]

\[P(\mu-2\sigma\le X\le\mu+2\sigma)\approx 0,95\]

\[P(\mu-3\sigma\le X\le\mu+3\sigma)\approx 0,997\]
\end{minipage}
\begin{minipage}{0.5\linewidth}
\includegraphics[width=0.95\linewidth]{loi_normale.png}
\end{minipage}
\end{propriete}

\begin{easybox}[Approximation de la loi binomiale par la loi normale]{}

	
{\it Lorsque $n$ est grand et $p$ ni trop près de $0$, ni trop près de $1$, alors la loi binomiale est très proche d'une loi normale ayant même espérance et même écart-type.

En pratique, on convient de faire cette approximation pour $n \geq 50$, $p$ ni proche de $0$ ni de $1$ et $np(1 - p) > 10$.}


Dans les bonnes conditions, on peut approcher un loi binomiale $\mathscr{B}(n,p)$ par une loi normale $\mathscr{N}(\mu,\sigma)$ avec $\mu=np$ et $\sigma=\sqrt{np(1-p)}$


\begin{center}
	\includegraphics[width=0.99\linewidth]{approxim.png}
\end{center}

\end{easybox}

\subsubsection{Exemple et correction de continuité}

Soit $X$ une variable aléatoire suivant la loi  $\mathcal{B}(50;0,4)$.

On peut approcher cette loi binomiale par une loi normale de moyenne $\mu=50\times0,4=20$ et d'écart type $\sigma=\sqrt{50\times0,4\times(1-0,4)}\approx3,4641$.

Pour calculer une probabilité, il faut alors tenir compte de la correction de continuité :
\begin{itemize}

	\item[$\bullet$]  pour approcher $P(X\le 22)$, on calcule $P(Y\le 22,5)$,
	\item[$\bullet$]  pour approcher $P(X\geq 16)$, on calcule $P(Y\geq 15,5)$,
	\item[$\bullet$]  pour approcher $P(18\le X\le 22)$, on calcule $P(17,5\le Y\le 22,5)$.
\end{itemize}

\begin{definition}[La loi uniforme]{}
Une \textbf{loi de probabilité} est dite \textbf{uniformee} si tous les éléments de l'univers ont la même probabilité d'occurrence.
	
\medskip

Une variable aléatoire $X$ suit une loi uniforme sur l'intervalle $\ff{a}{b}$ notée $X \sim \mathcal{U}\ff{a}{b}$ lorsque la densité de probabilité de $f$ est définir par:\\

\begin{minipage}{8cm}
$$f(x)=
\begin{cases}
\dfrac{1}{b-a} & \textrm{si} \qquad a \leqslant x \leqslant b \\
0 & \textrm{sinon}
\end{cases}$$

\end{minipage}
\begin{minipage}{8cm}
$$P(c \leqslant X \leqslant d)=\dfrac{d-c}{b-a}$$
\end{minipage}
\begin{center}
	\includegraphics[width = 0.6\linewidth]{C220_1.jpg}
	
\end{center}
\end{definition}

\begin{propriete}[Propriétés des lois uniformes]{}
Soit $X$ une variable aléatoire suivant une loi uniforme sur l'intervalle $\ff{a}{b}$.

\begin{itemize}
	\item[$\bullet$] L'espérance de $X$ est : $E(X)=\dfrac{a+b}{2}$.
	\item[$\bullet$] La variance de $X$ est : $V(X)=\dfrac{(b-a)^2}{12}$.
	\item[$\bullet$] L'écart-type de $X$ est : $\sigma(X)=\dfrac{b-a}{\sqrt{12}}$.

\end{itemize}
\end{propriete}

\section{Lois discrètes}

\begin{propriete}[Lois discrètes]{}
Une \textbf{loi de probabilité} est dite \textbf{discrète} si elle est définie sur un ensemble fini ou dénombrable de valeurs.

On peut donc compter les éléments de l'univers et attribuer une probabilité à chacun d'eux. Si on jette un dé équilibré à six faces, la loi de probabilité associée est discrète car elle est définie sur l'ensemble fini $\{1, 2, 3, 4, 5, 6\}$.
\end{propriete}

\begin{exemple}[\quad]{}
	\begin{itemize}
		\item La \textbf{loi de Bernoulli} est une loi de probabilité discrète qui modélise une expérience aléatoire binaire, c'est-à-dire une expérience qui n'a que deux issues possibles: succès ou échec.
		\item La \textbf{loi binomiale} est une loi de probabilité discrète qui modélise le nombre de succès dans une séquence d'expériences de Bernoulli indépendantes.
	\end{itemize}
\end{exemple}


\section{Lois continues}

\begin{definition}[Lois continues]{}
Une \textbf{loi de probabilité} est dite \textbf{continue} si elle est définie sur un intervalle de valeurs réelles. 

\begin{itemize}
	\item les lois continues ne sont pas définies sur des points individuels,
	\item elles sont plutôt définies sur des intervalles de valeurs.
\end{itemize}
\end{definition}

\begin{exemple}[Exemples de lois continues]{}
	\begin{itemize}
		\item on définit une variable aléatoire qui associe à chaque train, le temps de trajet entre deux gares.
		\item on définit une variable aléatoire qui associe à chaque personne, sa taille.
		\item on définit une variable aléatoire qui associe à chaque jour, la température maximale enregistrée dans une ville.
	\end{itemize}
\end{exemple}

\begin{easybox}[Fonction de densité]{}
Une \textbf{fonction de densité} est une fonction qui permet de définir une loi de probabilité continue. Elle doit satisfaire les conditions suivantes:

\medskip

\begin{minipage}{0.7\linewidth}
	
\begin{itemize}
	\item $f(x) \geq 0$ pour tout $x \in \R$ et  $\displaystyle\int_{-\infty}^{+\infty} f(x) \, dx = 1$
	\item pour tout intervalle $\ff{a}{b}$, la probabilité que la variable aléatoire prenne une valeur dans cet intervalle est donnée par l'intégrale de la fonction de densité sur cet intervalle: \[P(X \in \ff{a}{b}) = P(a \leq X \leq b)= \int_{a}^{b} f(x) \, dx\]
\end{itemize}
\end{minipage}\hfill
\begin{minipage}{0.3\linewidth}
\includegraphics[width = 0.9\linewidth]{loiproba2.png}
\end{minipage}
\end{easybox}

\pagebreak

\begin{exemple}[Exemple de fonction de densité]{}
\begin{enumerate}
	\item Vérifier que la fonction $f$ définie ci-dessous est une fonction de densité.
	$$f(x) = \begin{cases}
2\e^{-x} & \text{si } x \in \ff{0}{\ln(2)}  \\
0 & \text{sinon}
\end{cases}$$
	\item Calculer $\displaystyle\int_{0}^{0,5} f(x) \, dx$ et interpréter le résultat en terme de probabilité.
	\item Calculer la probabilité que la variable aléatoire prenne une valeur supérieure à $0,5$.
	\item Calculer la probabilité que la variable aléatoire prenne une valeur comprise entre $0,2$ et $0,5$.
	\item Créer un curseur $a$ dans l'intervalle $\ff{0}{\ln(2)}$ et 
	\begin{itemize}
		\item déterminer $a$ tel que $P(X \leq a) = 0,25$,
		\item déterminer $a$ tel que $P(X \leq a) = 0,5$.
	\end{itemize}
\end{enumerate}
\end{exemple}

\begin{propriete}[Propriétés des lois continues]{}
\begin{itemize}
	\item $P(X=k)=\displaystyle\int_k^k\;f(x)\,dx=0$ 
	\item $P(X \leq a) = P(X < a)$ (la probabilité que la variable aléatoire prenne une valeur inférieure ou égale à $a$ est égale à la probabilité qu'elle prenne une valeur strictement inférieure à $a$)
	\item L'espérance d'une variable aléatoire continue $X$ est donnée par la formule: \[E(X) = \displaystyle\int_{-\infty}^{+\infty} x f(x) \, dx\]
\end{itemize}
\end{propriete}
