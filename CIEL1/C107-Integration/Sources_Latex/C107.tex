\section{Intégrale d'une fonction continue et positive}

\begin{definition}[Notion d'intégrale]
    Soit $f$ une fonction continue et positive sur un intervalle $\ff{a}{b}$ de courbe représentative $\mathcal{C}_f$ dans un repère orthogonal \Oij.
    \smallskip

    \begin{minipage}{0.75\linewidth}
        L'\textbf{intégrale} de $a$ à $b$ de $f$ est l'aire, exprimée en unités d'aires, du domaine situé entre la courbe $\mathcal{C}_f$, l'axe des abscisses et les droites d'équation $x=a$ et $x=b$.

        Cette aire se note $\boxed{\displaystyle\int_{a}^{b} f(x)\dx}$ et se lit \og{} intégrale de $a$ à $b$ de $f(x)$ $dx$ \fg{}.
    \end{minipage}\hfill
    \begin{minipage}{0.22\linewidth}
        \includegraphics[width=1\linewidth]{02}
    \end{minipage}
\end{definition}

\begin{remarque}
\begin{itemize}
    \item $a$ et $b$ sont les \textbf{bornes inférieure et supérieure} de l'intégrale. Par défaut $a \leq b$.
    \item $x$ est une \textbf{variable muette}. On peut remplacer $x$ par $y$, $t$\ldots
    \item Pour toute fonction positive: $\displaystyle\int_{a}^{a} f(x)\dx = 0$.
    \item Le symbole $\int$ (S allongé) est dû à G.W. Leibniz, rappelant qu'une aire est une somme d'aires élémentaires.
\end{itemize}
\end{remarque}



\begin{exemple}[Calcul d'aire]
    \begin{minipage}{0.80\linewidth}
        Soit $f$ la fonction définie par $f(x)=\dfrac{x}{2}+2$. On cherche à évaluer le domaine hachuré. Le domaine colorié est un trapèze dont l'aire est: \dotfill{}
        
        Les unités graphiques sont $OI=0,6$ cm et $OJ=0,8$ cm. Quelle est la valeur de l'aire coloriée?
    \end{minipage}\hfill

\begin{propriete}[Calcul d'une intégrale]
    Soit $f$ une fonction continue et positive sur $\ff{a}{b}$ et $F$ une primitive de $f$ sur cet intervalle, alors:
    \[ \boxed{\int_{a}^{b} f(x) \dx = {\Bigl[F(x)\Bigr]}_{a}^{b} = F(b) - F(a)} \]
\end{propriete}

\begin{methode}[Calculer une intégrale]
    Pour calculer $\int_{a}^{b} f(x) \dx$:
    \begin{enumerate}
        \item Identifier les bornes $a$ et $b$.
        \item S'assurer que $f$ est continue sur $\ff{a}{b}$.
        \item Déterminer une primitive $F$ de $f$.
        \item Calculer $F(b) - F(a)$.
    \end{enumerate}
\end{methode}

\section{Intégration d'une fonction continue de signe quelconque}

\begin{propriete}[Cas des fonctions non positives]
    Si la fonction $f$ n'est pas positive sur tout l'intervalle $\ff{a}{b}$, alors on ne peut PLUS dire que $\int_{a}^{b}f(x)dx$ est l'aire sous la courbe représentative de la fonction $f$.
\end{propriete}

\begin{propriete}[Propriétés de l'intégrale]
    Pour toutes fonctions $f$ et $g$ continues sur $\ff{a}{b}$:
    \small
    \[ \int_{b}^{a} f(x)\dx = -\int_{a}^{b}f(x)\dx \qquad ; \qquad \int_{a}^{b}\Bigl(f+g\Bigr)(t)\dt = \int_{a}^{b} f(t)\dt + \int_{a}^{b}g(t)\dt \]
\end{propriete}

\medskip

\begin{propriete}[Fonction négative et aire]
    \begin{minipage}{0.6\linewidth}
        Soit $f$ une fonction continue est négative sur un intervalle $\ff{a}{b}$. Alors l'aire du domaine situé entre $\mathcal{C}_f$ et l'axe des abscisses sur $\ff{a}{b}$ est 
        \medskip
        \[A_\mathcal{D}= A_\mathcal{E}=\int_{a}^{b}(-f(x))\dx=-\int_{a}^{b}f(x)\dx\]

\section{Valeur moyenne}

\begin{definition}[Valeur moyenne]
    Soit $f$ une fonction continue sur un intervalle $\ff{a}{b}$. La \textbf{valeur moyenne} de $f$ sur $\ff{a}{b}$ est le nombre $\mu$ défini par:
    \[\boxed{\mu=\dfrac{1}{b-a}\int_{a}^{b} f(t)\dt}\]
\end{definition}


\begin{remarque}
    \begin{minipage}{0.59\linewidth}
        Dans le cas où $f$ est positive et continue sur $\ff{a}{b}$, la
        valeur moyenne de $f$ entre $a$ et $b$ représente la hauteur
        du rectangle construit sur l'intervalle $\ff{a}{b}$.\\
        L'aire du rectangle ABCD est égale, en u.a., à l'aire du
        domaine coloré car d'après la définition:
        \[\mu\cdot(b-a)=\int_{a}^{b}f(t)\dt\]
