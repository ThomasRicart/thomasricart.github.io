\begin{definition}[Notion de complexité d'un algorithme]{}
	La \textbf{complexité} d'un algorithme est une mesure de la quantité de ressources (temps, mémoire, etc.) que l'algorithme utilise en fonction de la taille de l'entrée. La complexité permet de comparer les algorithmes entre eux et de choisir le plus efficace pour résoudre un problème donné.	

	Nous allons nous intéresser à la complexité temporelle, (temps d'exécution de l'algorithme en fonction de la taille de l'entrée) exprimée en utilisant la notation \textbf{O} (grand O).
	
	Pour étudier la complexité d'un algorithme on se place toujours dans le \textbf{pire des cas}, c'est à dire que l'on suppose que l'algorithme doit parcourir tous les éléments de l'entrée pour trouver la solution.
\end{definition}

\begin{easybox}{Complexité de la recherche naïve}{}
	
\begin{itemize}
	\item Dans le pire des cas, l'algorithme de recherche naïve doit parcourir tous les éléments de la liste pour vérifier si l'élément recherché est présent ou non. (Elément absent ou présent à la fin de la liste)
	\item Si la liste possède n éléments, alors l'algorithme de recherche naïve doit effectuer n comparaisons. On dira que la complexité est de l'ordre de O(n).
	\item Si la liste contient 10 éléments, l'algorithme doit effectuer 10 comparaisons dans le pire des cas. 
	\item De manière générale, si la taille de la liste double, le temps d'exécution de l'algorithme de recherche naïve double également. La complexité de l'algorithme de recherche naïve est linéaire.
\end{itemize}
	
\end{easybox}


\begin{definition}[Notion de terminaison d'un algorithme]{}
	Un algorithme est dit \textbf{terminant} s'il s'arrête après un nombre fini d'étapes pour toute entrée donnée. 

	Un algorithme se termine de façon certaine si:
\begin{itemize}
\item Il ne contient pas de boucle infinie.
\item S'il fait appel à une boucle infinie (WHILE), il doit y avoir une condition d'arrêt qui est vérifiée à un moment donné.
\end{itemize}

La condition d'arrêt est appelée \textbf{variant de boucle}.	
\end{definition}

\pagebreak

\begin{definition}[Le variant de boucle]{}
	Un \textbf{variant de boucle} est une quantité qui est modifiée à chaque itération d'une boucle et qui permet de garantir la terminaison de l'algorithme. 

	Pour qu'un variant de boucle soit efficace, il doit être:
\begin{itemize}
	\item \textbf{positif ou nul} pour que la boucle se poursuive;
	\item \textbf{décroissant} à chaque itération pour que la boucle se rapproche de sa condition d'arrêt.
\end{itemize}
\end{definition}

\begin{definition}[Le logarithme]
	Le \textbf{logarithme} est un opérateur mathématique qui a entre autre comme propriété:

	\[
	\log_2(2^k) = k \times \log_2(2)=k \qquad \textrm{ et } \log_2(2) = 1
	\]
\end{definition}

	Dans notre exemple, lorsque la taille de la liste double, alors le nombre de boucles augmente seulement de 1.

	Supposons que la liste comporte $n$ éléments et que $n$ est tel qu'il existe un $k\in\N$ tel que $n=2^k$.

	Etant donné que le nombre de boucles augmente de 1 quand le nombre d'éléments est multiplié par deux alors $k$ est le nombre de boucles nécessaires pour obtenir une liste d'un seul élément.
	
\smallskip

	\begin{minipage}{0.39\linewidth}
		\begin{center}
		
	\begin{tabular}{c|c|c}
		$n$ & valeur de $k$ & nombre de boucles \\\hline
		1 & 0 & 0 \\
		2 & 1 & 1 \\
		4 & 2 & 2 \\
		8 & 3 & 3 \\
		16 & 4 & 4 \\
		\ldots{} & \ldots{} & \ldots{} \\
		n & k & k \\
		2n & k+1 & k+1 \\
		
	\end{tabular}
\end{center}
	\end{minipage}\hfill{}
	\begin{minipage}{0.59\linewidth}
Pour extraire $k$ en fonction de $n$, on utilise une fonction nommée \verb+logarithme+ et notée \verb+log+:


\[n = 2^k \Longleftrightarrow \log_2(n)=\log_2(2^k) = k\log_2(2)=k\]

Ainsi $k = \log_2(n)$

$k$ étant le nombre de boucles nécessaires pour obtenir une liste d'un seul élément, alors la complexité est de l'ordre de $O(\log(n))$.
	
	\end{minipage}
	



\begin{definition}[Complexité logarithmique]{}

Dans l'exemple précédent, on dira que cet algorithme possède une complexité \textbf{logarithmique} notée $O(\log(n))$

	
\end{definition}

\begin{propriete}[Détermination de complexité]{}
	Pour déterminer la complexité d'un algorithme, on peut utiliser les propriétés suivantes:

	\begin{itemize}
		\item Si un algorithme n'effectue pas de boucle, alors sa complexité est de l'ordre de $O(1)$.
		\item Si un algorithme effectue une boucle FOR, alors sa complexité est de l'ordre de $O(n)$.
		\item Si un algorithme effectue k boucles FOR imbriquées, alors sa complexité est de l'ordre de $O(n^k)$.
		\item Si un algorithme effectue une boucle WHILE et à chaque itération la taille du problème est divisée par 2, alors sa complexité est de l'ordre de $O(\log(n))$.
	\end{itemize}
\end{propriete}

\section{Les listes}


\begin{definition}[Les listes]{}
	Une \textbf{liste} est une structure de donnée qui contient des éléments ou items identifiés par des indices.

Ces éléments sont:
\begin{itemize}
\item \textbf{ordonnés}: les éléments d'une liste sont rangés dans un ordre précis. Chaque élément a une position dans la liste, appelée indice.
\item \textbf{séparés par des virgules}: les éléments d'une liste sont séparés par des virgules.
\item \textbf{entre crochets}: les éléments d'une liste sont entourés de crochets \verb|[ ]| en python.
\end{itemize}


\medskip

\hline

Elément & \verb|T[0]| & \verb|T[1]| & \verb|T[2]| & \verb|T[3]| & \verb|T[4]| & \verb|T[5]| & \verb|T[6]| & \verb|T[7]| \\ \hline

\end{tabular}



\end{definition}

\begin{easybox}{Quelques méthodes sur les listes}{}

\begin{itemize}
\item \verb|append(elt)|: ajoute l'élément \verb|elt| à la fin de la liste	
\item \verb|pop()|: supprime et renvoie le dernier élément de la liste
\item \verb|pop(i)|: supprime et renvoie l'élément d'indice \verb|i| de la liste
\end{itemize}

\begin{ConsolePythontex}[TaillePolice=\large, Label=false]{}
ma_liste = [1, 2, 3, 4, 5]
ma_liste.append(6)
print(ma_liste)
a = ma_liste.pop()
print(a)
print(ma_liste)
b = ma_liste.pop(2)
print(b)
print(ma_liste)
		\end{ConsolePythontex}
\end{easybox}

\begin{easybox}{Le hasard et les listes}{}
	\begin{itemize}
		\item La fonction \verb|random.choice(liste)| permet de choisir un élément au hasard dans une liste.
		\item La fonction \verb|random.shuffle(liste)| permet de mélanger les éléments d'une liste de manière aléatoire.
	\end{itemize}

	\begin{ConsolePythontex}[TaillePolice=\large, Label=false]{}
import random
ma_liste = [1, 2, 3, 4, 5]
print(random.choice(ma_liste))
print(ma_liste)
random.shuffle(ma_liste)
print(ma_liste)
		\end{ConsolePythontex}
\end{easybox}

\begin{easybox}{Le slicing}{}
	Le slicing permet de créer une nouvelle liste à partir d'une liste existante en sélectionnant une portion de cette liste.

	\begin{CodePythontexAlt}[TaillePolice=\large, Largeur=1\linewidth,Centre,PremLigne=1]{}
ma_liste = [1, 2, 3, 4, 5]
print(ma_liste[1:4])
# Affiche: [2, 3, 4]
print(ma_liste[:3])
# Affiche: [1, 2, 3]
print(ma_liste[2:])
# Affiche: [3, 4, 5]
		\end{CodePythontexAlt}
\end{easybox}

\begin{easybox}{Les effets de bord sur les listes}{}
	\begin{itemize}
		\item Les listes sont des objets mutables, c'est à dire que leur contenu peut être modifié après leur création.
		\item Lorsque l'on affecte une liste à une variable, on crée une référence à cette liste. Si on modifie la liste via cette variable, la modification sera visible via toutes les variables qui font référence à cette liste.
	\end{itemize}

	\begin{ConsolePythontex}[TaillePolice=\large, Label=false]{}
ma_liste = [1, 2, 3, 4, 5]
ma_liste2 = ma_liste
ma_liste2[0] = 10
print(ma_liste)
print(ma_liste2)
		\end{ConsolePythontex}
\end{easybox}

Si on veut faire une vraie copie d'une liste, on peut utiliser la syntaxe suivante:
	\begin{ConsolePythontex}[TaillePolice=\large, Label=false]{}
ma_liste = [1, 2, 3, 4, 5]
ma_liste2 = ma_liste[:]
ma_liste2[0] = 10
print(ma_liste)
print(ma_liste2)
		\end{ConsolePythontex}

Lorsqu'on utilise des listes dans des fonctions, il faut faire attention aux effets de bord. Si on modifie une liste passée en argument d'une fonction, la modification sera visible à l'extérieur de la fonction.

Dans cet exemple, il n'est pas utile de renvoyer la liste modifiée, car la modification est déjà visible à l'extérieur de la fonction.
\begin{CodePythontexAlt}[TaillePolice=\large, Largeur=1\linewidth,Centre,PremLigne=1]{}	
def ajoute_element(liste, element):
	liste.append(element)

ma_liste = [1, 2, 3, 4, 5]
ajoute_element(ma_liste, 6)
print(ma_liste)			# Affiche: [1, 2, 3, 4, 5, 6]
		\end{CodePythontexAlt}

		\section{Les tuples}

		\begin{definition}[Les tuples]{}
		Un \textbf{tuple} est une structure de donnée qui contient des éléments ou items identifiés par des indices.	
		Les éléments d'un tuple sont:
\begin{itemize}
\item \textbf{ordonnés}: chaque élément a une position dans le tuple, appelée indice.
\item \textbf{séparés par des virgules}: les éléments d'un tuple sont séparés par des virgules.
\item \textbf{entre parenthèses}: les éléments d'un tuple sont entourés de parenthèses \verb|( )| en python.
\end{itemize}
\medskip
\begin{tabular}{|p{2cm}|p{1cm}|p{1cm}|p{1cm}|p{1cm}|p{1cm}|p{1cm}|p{1cm}|p{1cm}|}
\hline
Indice \verb|i| & 0 & 1 & 2 & 3 & 4 & 5 & 6 & 7 \\ \hline
Elément	& \verb|T[0]| & \verb|T[1]| & \verb|T[2]| & \verb|T[3]| & \verb|T[4]| & \verb|T[5]| & \verb|T[6]| & \verb|T[7]| \\ \hline

\end{tabular}

			
		\end{definition}

\begin{easybox}{Accéder aux éléments d'un tuple}{}
	Pour accéder à un élément d'un tuple, on utilise son indice. Les indices commencent à 0.

	\smallskip

	\begin{minipage}{0.48\linewidth}
		\begin{ConsolePythontex}[TaillePolice=\large, Label=false]{}

print(mon_tuple[0])

		\end{ConsolePythontex}
	\end{minipage}\hfill
	\begin{minipage}{0.48\linewidth}
		\begin{ConsolePythontex}[TaillePolice=\large, Label=false]{}

		\end{ConsolePythontex}
	\end{minipage}
	\end{easybox}	

	\begin{propriete}[Immutabilité des tuples]{}
		Les éléments d'un tuple ne peuvent pas être modifiés après la création du tuple. Un tuple est une structure de donnée immuable.

	\end{propriete}
