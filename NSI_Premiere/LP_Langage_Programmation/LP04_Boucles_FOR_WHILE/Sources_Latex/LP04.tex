\section{La boucle FOR}

\begin{easybox}{Les boucles FOR}{}

	La boucle \verb+FOR+ est une boucle bornée. On sait à l'avance combien de fois on va la faire.

\medskip

	\begin{minipage}{0.35\linewidth}
	\begin{lstlisting}[numbers=none]
# En pseudo code
instruction1
Pour i allant de 0 a 6:
	instruction2
instruction3
	\end{lstlisting}
		\end{minipage}\hfill
		\begin{minipage}{0.63\linewidth}
	\begin{itemize}
		\item instruction1 est effectuée
		\item \verb+i+ est un \verb+itérateur+ qui va aller de 0 à 6, c'est à dire 7 tours
		\item On effectuera donc 7 fois l'instruction2 
		\item Quand on a fini la boucle on effectue l'instruction3
	\end{itemize}
		\end{minipage}	
	\end{easybox}

	\begin{easybox}{Prototypage de la boucle FOR}
	La boucle FOR s'écrit de manière générale de la façon suivante:
	
	\medskip
	
	\begin{minipage}[b]{0.4\linewidth}
	\begin{CodePythontexAlt}[TaillePolice=\large, Largeur=1\linewidth,Centre,PremLigne=1]{}
		for i in range(a, b):
	\end{CodePythontexAlt}
	\end{minipage}\hfill
	\begin{minipage}[b]{0.58\linewidth}
	La boucle est effectuée $b-a$ fois et la variable \verb+i+ prend pour valeur tous les entiers entre $\fo{a}{b}$
	\end{minipage}
	
	
	\end{easybox}
	

	\begin{minipage}{0.45\linewidth}
		
	\begin{exemple}[Trace de la structure FOR]{}
		\begin{CodePythontexAlt}[TaillePolice=\large, Largeur=1\linewidth,Centre,PremLigne=1]{}
		for i in range(2, 6):
			print(i)
		print('fin')
	\end{CodePythontexAlt}
	
		\end{exemple}

	\end{minipage}\hfill{}

	\begin{easybox}{Itération sur une chaine de caractères via les éléments}{}

	Il est aussi possible	d'itérer directement sur les éléments qui constituent cette chaine de caractère
	
	On peut donc créer un variable nommée \verb+lettre+ par exemple qui prendra pour valeur chacun des caractères de \verb+mot+.
	
	\medskip
	
	\begin{minipage}{0.48\linewidth}
	\begin{CodePythontexAlt}[TaillePolice=\large, Largeur=1\linewidth,Centre,PremLigne=1]{}
	mot = 'Hello'
	for lettre in mot:
		print(lettre)
	print('Fin')
	
		\end{CodePythontexAlt}
	\end{minipage}\hfill
	\begin{minipage}{0.48\linewidth}
		\begin{center}
			\renewcommand{\arraystretch}{1}
			\begin{tabular}{|c|c|c|c|}
				\hline
				lettre  & instruction & affichage \\ \hline{}
				\verb+'H'+  & \verb+print(lettre)+ & \verb+H+ \\ \hline{}
				\verb+'e'+  & \verb+print(lettre)+ & \verb+e+ \\ \hline{}
				\verb+'l'+  & \verb+print(lettre)+ & \verb+l+ \\ \hline{}
				\verb+'l'+  & \verb+print(lettre)+ & \verb+l+ \\ \hline{}
				\verb+'o'+  & \verb+print(lettre)+ & \verb+o+ \\ \hline{}
				\verb+None+  & \verb+print('Fin')+ & \verb+Fin+ \\ \hline
			\end{tabular}	
		\end{center}
	
	
	\end{minipage}
	\begin{itemize}
		\item Il n'y a pas besoin de connaitre la longueur du mot
		\item Il n'y  plus l'utilisation d'indice
		\item \verb+lettre+ prend pour valeur chaque caractère de mot
		\item lorsque \verb+lettre+ a pris pour valeur tous les caractères de \verb+mot+, la boucle s'arrête.
	\end{itemize}								
	\end{easybox}

	\begin{easybox}{Algorithmes de Recherche dans un itérable}
	
		Cet algorithme permet de savoir si un élément est présent dans un itérable. 

			\begin{itemize}

				\item Le mot sera la variable \verb|mot|
				\item La lettre que l'on cherche sera la variable \verb|lettre_cherchee|
			\end{itemize}

			On peut utiliser l'algorithme suivant:
			
	\begin{enumerate}
		\item on parcourt \verb|mot| à l'aide d'une boucle \verb|FOR|
		\item on observe donc chaque caractère \verb|mot[i]|
		\item si le caractère est égal à \verb|lettre_cherchee|, alors on renvoie la valeur \verb|True| et on arrête la boucle.
		\item sinon on ne fait rien et on revient à l'étape 2 pour observer le caractère suivant.
		\item si on a observé tous les caractères de \verb|mot| sans trouver \verb|lettre_cherchee|, on renvoie \verb|False|
	\end{enumerate}
		\end{easybox}

		\section{La boucle WHILE}

		\begin{easybox}{Les boucles WHILE}
		La boucle \verb+WHILE+ est une boucle non bornée. On ne sait à l'avance combien de fois on va la faire.
	
		\medskip
	\begin{minipage}{0.4\linewidth}
	\begin{lstlisting}[numbers=none]
# En pseudo code
instruction1
tant que condition1 vraie:
	instruction2
instruction3
	\end{lstlisting}
		\end{minipage}\hfill
		\begin{minipage}{0.59\linewidth}
	\begin{itemize}
		\item instruction1 est effectuée
		\item tant que la condition1 est vraie on effectue l'instruction 2
		\item quand la condition1 est fausse, on sort de la boucle;
		\item on effectue l'instruction3
	\end{itemize}
		\end{minipage}	
	\end{easybox}
	

		\begin{easybox}{Prototypage de la boucle WHILE en python}
	\verb|condition1| est un booléen qui est évalué à chaque tour de boucle.  Tant que \verb|condition1| \verb|True|, on effectue les instructions de la boucle. Lorsque \verb|condition1| devient \verb|False|, on sort de la boucle.

	\begin{center}
	
	\begin{CodePythontexAlt}[TaillePolice=\large, Largeur=1\linewidth,Centre,PremLigne=1]{}
		while condition1:
			instruction
	\end{CodePythontexAlt}
	\end{center}
	

			\end{easybox}

	\begin{exercice_cours}[Dire ce qu'affichent les programmes suivants et expliquer pourquoi]{}
		\begin{minipage}{0.48\linewidth}
			\begin{CodePythontexAlt}[TaillePolice=\large, Largeur=1\linewidth,Centre,PremLigne=1]{}

			u = 1
			while u < 1000:
				n = n + 1
				u = u * 2
			print(n)
			print(u)
		\end{CodePythontexAlt}
				
		\end{minipage}\hfill
		\begin{minipage}{0.48\linewidth}
			\begin{CodePythontexAlt}[TaillePolice=\large, Largeur=1\linewidth,Centre,PremLigne=1]{}

			c = 0
			trouve = False
			while c < len(mot) and not(trouve):
				trouve = mot[c] == 'r'
				c = c + 1
			print(trouve)
			\end{CodePythontexAlt}
		\end{minipage}\hfill
		 
	\end{exercice_cours}
