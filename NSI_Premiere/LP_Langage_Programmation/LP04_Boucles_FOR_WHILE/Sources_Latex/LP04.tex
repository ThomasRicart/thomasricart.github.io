\begin{easybox}{Les boucles FOR}{}

	La boucle \verb+FOR+ est une boucle bornée. On sait à l'avance combien de fois on va la faire.

\medskip

	\begin{minipage}[b]{0.35\linewidth}
	\begin{lstlisting}[numbers=none]
# En pseudo code
instruction1
Pour i allant de 0 a 6:
	instruction2
instruction3
	\end{lstlisting}
		\end{minipage}\hfill
		\begin{minipage}[b]{0.63\linewidth}
	\begin{itemize}
		\item instruction1 est effectuée
		\item \verb+i+ est un \verb+itérateur+ qui va aller de 0 à 6, c'est à dire 7 tours
		\item On effectuera donc 7 fois l'instruction2 
		\item Quand on a fini la boucle on effectue l'instruction3
	\end{itemize}
		\end{minipage}	
	\end{easybox}

	\end{easybox}
	
	
	\begin{easybox}{Prototypage de la boucle FOR}
	La boucle FOR s'écrit de manière générale de la façon suivante:
	
	\medskip
	
	\begin{minipage}[b]{0.4\linewidth}
	\begin{CodePythontexAlt}[TaillePolice=\large, Largeur=1\linewidth,Centre,PremLigne=1]{}
		for i in range(a, b):
	\end{CodePythontexAlt}
	\end{minipage}\hfill
	\begin{minipage}[b]{0.58\linewidth}
	La boucle est effectuée $b-a$ fois et la variable \verb+i+ prend pour valeur tous les entiers entre $\fo{a}{b}$
	\end{minipage}
	
	
	\end{easybox}
	
	\begin{exemple}[Algorigramme de la structure FOR]{}
	
	
	\begin{center}
	{\verb+for i in range(2, 9):+}
		
	\end{center}
	
	\tcblower{
	
	\begin{CodePythontexAlt}[TaillePolice=\large, Largeur=1\linewidth,Centre,PremLigne=1]{}
		for i in range(2, 9):
			print(i)
		print('fin')
	\end{CodePythontexAlt}
	
	\medskip
	
	
	
	L'itérateur \verb+i+ prend toutes les valeurs entières entre 2 et $9-1$

	\begin{center}

		\verb+3+  & \verb+True+ & \verb+print(3)+ \\ \hline{}
		\verb+4+  & \verb+True+ & \verb+print(4)+ \\ \hline{}
		\verb+5+  & \verb+True+ & \verb+print(5)+ \\ \hline{}
		\verb+6+  & \verb+True+ & \verb+print(6)+ \\ \hline{}
		\verb+7+  & \verb+True+ & \verb+print(2)+ \\ \hline{}
		\verb+8+  & \verb+True+ & \verb+print(2)+ \\ \hline{}
		\verb+9+  & \verb+False+ & \verb+print('fin')+ \\\hline
		
	\end{tabular}	
	\end{center}
	
	}
	\end{exemple}

	Une chaine de caractères \verb+str+ est un itérateur, c'est à dire qu'il est possible d'itérer sur ses éléments.
	
	\begin{minipage}[b]{0.48\linewidth}

	mot[1]
	mot[3]

	\begin{minipage}[b]{0.48\linewidth}
		\begin{center}

			\hline

				\verb+1+  & \verb+True+ & \verb+print(mot[1])+ & \verb+e+ \\ \hline{}

	\begin{easybox}{Itération sur une chaine de caractères via les éléments}{}
	Une chaine de caractères \verb+str+ est un itérateur, c'est à dire qu'il est possible
	d'itérer directement sur les éléments qui constituent cette chaine de caractère
	
	On peut donc créer un variable nommée \verb+lettre+ par exemple qui prendra pour valeur chacun des caractères de \verb+mot+.
	
	\medskip
	
	\begin{minipage}[b]{0.48\linewidth}
	\begin{CodePythontexAlt}[TaillePolice=\large, Largeur=1\linewidth,Centre,PremLigne=1]{}
	mot = 'Hello'
	for lettre in mot:
		print(lettre)
	print('Fin')
	
		\end{CodePythontexAlt}
	\end{minipage}\hfill
	\begin{minipage}[t]{0.48\linewidth}
		\begin{center}
			\renewcommand{\arraystretch}{1}
			\begin{tabular}{|c|c|c|c|}
				\hline
				lettre  & instruction & affichage \\ \hline{}
				\verb+'H'+  & \verb+print(lettre)+ & \verb+H+ \\ \hline{}
				\verb+'e'+  & \verb+print(lettre)+ & \verb+e+ \\ \hline{}
				\verb+'l'+  & \verb+print(lettre)+ & \verb+l+ \\ \hline{}
				\verb+'l'+  & \verb+print(lettre)+ & \verb+l+ \\ \hline{}
				\verb+'o'+  & \verb+print(lettre)+ & \verb+o+ \\ \hline{}
				\verb+None+  & \verb+print('Fin')+ & \verb+Fin+ \\ \hline
			\end{tabular}	
		\end{center}
	
	
	\end{minipage}
	\begin{itemize}
		\item Il n'y a pas besoin de connaitre la longueur du mot
		\item Il n'y  plus l'utilisation d'indice
		\item \verb+lettre+ prend pour valeur chaque caractère de mot
		\item lorsque \verb+lettre+ a pris pour valeur tous les caractères de \verb+mot+, la boucle s'arrête.
	\end{itemize}								
	\end{easybox}

	\begin{easybox}{Algorithmes de Recherche dans un itérable}
	
	Puisqu'on peut itérer sur une chaine de caractère en observant chaque caractère, on peut décider de faire quelque chose si la caractère a ue particularité.
	
	Ce algorithme compte le nombre de caractères \verb+'a'+ dans la chaine de caractère \verb+mot+
	
	\medskip
	
	\begin{minipage}[t]{0.55\linewidth}

			\draw [arrow] (initialisation_i.south) -- (decision1.north);
			\draw [arrow] (decision1.south) -- node[anchor=east]{False}(return.north);

	\begin{minipage}[b]{0.43\linewidth}
	
	  \begin{itemize}
			\item on initialise un compteur à 0
			\item on parcourt \verb+mot+ à l'aide d'une boucle \verb+FOR+
			\item on observe donc chaque caractère \verb+mot[i]+
			\item si le caractère est un \verb+'a'+, alors on incrémente le compteur
			\item sinon on ne fait rien 
			\item dans tous les cas on refait une boucle
		\end{itemize}

	def compte_a(mot):
		compteur = 0
		for i in range(len(mot)):
			if mot[i] == 'a':
				compteur=compteur+1
		return compteur
		
	\end{CodePythontexAlt}

				if mot[i] == caractere:
					compteur=compteur+1
			return compteur

		\section{La boucle WHILE}

		\begin{easybox}{Les boucles WHILE}
		La boucle \verb+WHILE+ est une boucle non bornée. On ne sait à l'avance combien de fois on va la faire.
	
	\begin{minipage}[b]{0.45\linewidth}
	\begin{lstlisting}[numbers=none]
	# En pseudo code
	instruction1
	tant que condition1 vraie:
		instruction2
	instruction3
	\end{lstlisting}
		\end{minipage}\hfill
		\begin{minipage}[b]{0.53\linewidth}
	\begin{itemize}
		\item instruction1 est effectuée
		\item tant que la condition1 est vraie on effectue l'instruction 2
		\item quand la condition1 est fausse, on sort de la boucle;
		\item on effectue l'instruction3
	\end{itemize}
		\end{minipage}	
	\end{easybox}
	
	\begin{easybox}{La boucle WHILE en python}
		\begin{minipage}[b]{0.48\linewidth}
			\begin{CodePythontexAlt}[TaillePolice=\large, Largeur=1\linewidth,Centre,PremLigne=1]{}
				a = 0
				while a < 5:
					print(a, end=" ")
					a = a + 1
				# affichage: 0 1 2 3 4
				\end{CodePythontexAlt}
				
		\end{minipage}\hfill
		\begin{minipage}[t]{0.48\linewidth}
			\begin{CodePythontexAlt}[TaillePolice=\large, Largeur=1\linewidth,Centre,PremLigne=1]{}
				lettre, mot = 'd', 'abracadabra'
				i = 0
				trouve = False
				while not trouve:
					trouve = (lettre == mot[i])
					i = i + 1
				print(i)
				# affichage 7
				# !!! possibilité de boucle infinie
			\end{CodePythontexAlt}
		\end{minipage}\hfill
		 
	\end{easybox}
	
	
	\begin{easybox}{Prototypage de la boucle WHILE}
	La boucle WHILE s'écrit de manière générale de la façon suivante:
	
	\medskip
	
	\begin{minipage}[b]{0.4\linewidth}
	\begin{CodePythontexAlt}[TaillePolice=\large, Largeur=1\linewidth,Centre,PremLigne=1]{}

			instruction
	\end{CodePythontexAlt}
	\end{minipage}\hfill
	\begin{minipage}[b]{0.58\linewidth}
	La boucle est effectuée tant que \verb|condition1| est vraie. Cette condition est donc un booléen\end{minipage}
	

	
	\end{easybox}

    \section{Les variables de type booléen \texttt{bool}}

    
\begin{easybox}{Les booléens}{}
    \smallskip
    Les booléens sont des variables qui ne peuvent prendre que deux valeurs: \verb+True+ (vrai) ou \verb+False+ (faux).
    
    \smallskip
    Les booléens peuvent être utilisés dans des opérations logiques.
    
	\medskip

    \begin{minipage}{0.32\linewidth}
        \begin{ConsolePythontex}[TaillePolice=\large, Label=false]{}

b = False
c = a and b

        \end{ConsolePythontex}
    \end{minipage}\hfill
    \begin{minipage}{0.32\linewidth}
        \begin{ConsolePythontex}[TaillePolice=\large, Label=false]{}

b = False
c = a or b
c
        \end{ConsolePythontex}
    \end{minipage}\hfill
    \begin{minipage}{0.32\linewidth}
        \begin{ConsolePythontex}[TaillePolice=\large, Label=false]{}
a = True
c = not a
c
        \end{ConsolePythontex}
    \end{minipage}    
\end{easybox}

\begin{easybox}{Booléen comme résultat de comparaison}{}
    Les booléens sont le résultat d'une comparaison. Les opérateurs de comparaison sont:
    \smallskip
    \begin{itemize}
        \begin{minipage}{0.44\linewidth}
            \item[] \texttt{==} \quad pour le test d'égalité
            \item[] \texttt{!=} \quad pour le test de différence
            \item[] \texttt{<} \quad pour le test d'infériorité
        \end{minipage}
        \begin{minipage}{0.44\linewidth}
            \item[] \texttt{>} \quad pour le test de supériorité
            \item[] \texttt{<=} \quad pour le test d'infériorité ou d'égalité
            \item[] \texttt{>=} \quad pour le test de supériorité ou d'égalité
        \end{minipage}
    \end{itemize}

    \end{easybox}
